# Create a jpeg

{% highlight js %}
var jpeg = require('jpeg-js');
var fs = require('fs');
var width = 320, height = 180;
var frameData = new Buffer(width * height * 4);
var i = 0;

var setColor = function(x, y, r, g, b, a) {

  var p = 4 * x + 4 * width * y;

  frameData[p++] = r;
  frameData[p++] = g;
  frameData[p++] = b;
  frameData[p++] = a;
}

console.log(0xFF)

for (var i = 0; i < width; i++) {
  for (var j = 0; j < height; j++) {
